import tokens from "@gemeente-rotterdam/design-tokens/dist/index.tokens.json";
import { ColorItem, ColorPalette, Markdown, Meta } from "@storybook/addon-docs";

export const baseColors = tokens["rods"]["color"]["base"];
export const grayTones = tokens["rods"]["color"]["gray"];
export const notificationColors = Object.entries(tokens["rods"]["color"]["notification"]);
export const skinColors = Object.entries(tokens["rods"]["color"]["skin"] || {});
export const extendedColors = Object.entries(tokens["rods"]["color"]).filter(
  ([key]) => !["base", "gray", "notification", "skin"].includes(key),
);
